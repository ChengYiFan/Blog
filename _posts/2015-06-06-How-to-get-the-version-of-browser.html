在IE11之前，我们可以通过以下方式获取IE浏览器的版本号：

    
        //return eg:6.0、7.0、8.0、9.0、10.0
        function getIEVer(){
            var agent = navigator.userAgent.toLowerCase();
            if(agent.indexOf("msie") > -1)
                return agent.match(/msie([\d.]+)/)[1];
        }
